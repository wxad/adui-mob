# StepsItem

用于显示任务进度或引导用户完成复杂任务的控件。

## Slot

export const slots = [
  {
    name: 'icon',
    desc: 'icon slot，自定义 icon',
  },
  {
    name: 'badge',
    desc: 'badge slot，自定义 badge 内容',
  },
  {
    name: 'desc',
    desc: '描述 slot，自定义描述内容',
  },
]

<Table
  columnsResizable
  dataSource={slots}
  columns={[
    {
      dataIndex: 'name',
      title: 'name',
    },
    {
      dataIndex: 'desc',
      title: '描述',
    },
  ]}
/>

<Alert style="margin-top: 20px;" title="注意事项" text="StepsItem 在数字类型使用的是 Badge 组件，所以，其对应的 slot 叫做 badge" />
